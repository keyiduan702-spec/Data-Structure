```
Data-Structure/
├── SList/      单链表
│   ├── SList.h
│   ├── SList.c
│   └── test.c
├── SeqList/    动态顺序表
│   ├── SeqList.h
│   ├── SeqList.c
│   └── test.c

    ├── Stack.c

```
